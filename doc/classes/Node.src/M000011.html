    def delete(element)
      if @keys.include?(element)       
        #Shift left
        sub_trees_old = @sub_trees.map
        ((@keys.index(element))..(@keys.size)-1).each do |i|
          @sub_trees[i] = sub_trees_old[i+1]
        end
        @sub_trees.pop
        @keys.delete(element)     
        true
      else
        false
      end
    end
